def linear_pred_model(data,test_data_size):
    ### Spliting of test and train data
    train_data,test_data = train_test_split(data,test_size=test_data_size)
    
    #### Assigning dependent and independent values from train data
    x_train = train_data['YearsExperience'].values.reshape(-1,1)
    y_train = train_data['Salary'].values.reshape(-1,1)
    
    ### Model creation
    model = LinearRegression()
    train_model = model.fit(x_train,y_train)
    print("Training of Model is Done...")
    
    #### Assigning dependent and independent values from test data
    x_test = test_data['YearsExperience'].values.reshape(-1,1)
    y_test = test_data['Salary'].values.reshape(-1,1)
    
    ### Predicting of accuracy 
    print("The accuracy by r2 score is..",round(r2_score(y_test,train_model.predict(x_test))*100,2),"% for the model with test size",test_data_size)

def pred_salary_hike(no_of_years):
    ### Spliting of test and train data
    train_data,test_data = train_test_split(data,test_size=0.2)
    
    #### Assigning dependent and independent values from train data
    x_train = train_data['YearsExperience'].values.reshape(-1,1)
    y_train = train_data['Salary'].values.reshape(-1,1)
    
    ### Model creation
    model = LinearRegression()
    train_model = model.fit(x_train,y_train)
    print("Training of Model is Done...")
    
    ### Storing the coefficient and intercept from the model to predict the output
    train_model_coef = train_model.coef_
    train_model_intercept = train_model.intercept_
    
    result = round(((train_model_coef * no_of_years) + train_model_intercept).flatten()[0],2)
    
    ### Printing the delivery time 
    print("The predicted Salary Hike is ...",result)

def pred_salary(no_of_years):
    ### Spliting of test and train data
    train_data,test_data = train_test_split(data,test_size=0.2)
    
    #### Assigning dependent and independent values from train data
    x_train = train_data['YearsExperience'].values.reshape(-1,1)
    y_train = train_data['Salary'].values.reshape(-1,1)
    
    ### Model creation
    model = LinearRegression()
    train_model = model.fit(x_train,y_train)
    
    return train_model.predict(no_of_years)
